    <style>
        body { font-family: 'Inter', sans-serif; background-color: #f3f4f6; }
        .theme-bg { background-color: #f59e0b; }
        .theme-text { color: #f59e0b; }
        .hide-scrollbar::-webkit-scrollbar { display: none; }
        .chat-bubble-me { background-color: #f59e0b; color: white; border-radius: 1.2rem 1.2rem 0 1.2rem; }
        .chat-bubble-other { background-color: #ffffff; color: #1f2937; border-radius: 1.2rem 1.2rem 1.2rem 0; border: 1px solid #e5e7eb; }

    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, addDoc, setDoc, deleteDoc, onSnapshot, collection, query, where, arrayUnion, updateDoc, getDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

        let app, db, auth;
        let userId = null;
        let currentView = 'market';
        let activeNegId = null;

        async function init() {
            app = initializeApp(firebaseConfig);
            db = getFirestore(app);
            auth = getAuth(app);
            
            if (initialAuthToken) {
                await signInWithCustomToken(auth, initialAuthToken).catch(() => signInAnonymously(auth));
            } else {
                await signInAnonymously(auth);
            }
            
            onAuthStateChanged(auth, async (user) => {
                if (user) {
                    userId = user.uid;
                    document.getElementById('my-uid').textContent = userId;
                    setupUserProfile();
                    switchView('market');
                }
            });
        }

        // --- Profile Logic ---
        async function setupUserProfile() {
            const userRef = doc(db, 'artifacts', appId, 'users', userId, 'profile', 'info');
            const snap = await getDoc(userRef);
            if (!snap.exists()) {
                await setDoc(userRef, { name: "ユーザー" + userId.substring(0,4), bio: "よろしくお願いします！", rating: 5.0 });
            }
        }

        // --- View Control ---
        window.switchView = function(view, data = null) {
            currentView = view;
            const views = ['market', 'negotiations', 'chat', 'profile'];
            views.forEach(v => document.getElementById(`${v}-view`).classList.add('hidden'));
            document.getElementById(`${view}-view`).classList.remove('hidden');
            
            if (view === 'market') loadItems();
            if (view === 'negotiations') loadNegotiations();
            if (view === 'chat' && data) initChat(data);
            if (view === 'profile') loadProfile();
        };

        // --- Market Logic ---
        function loadItems() {
            const itemsRef = collection(db, 'artifacts', appId, 'public', 'data', 'items');
            onSnapshot(itemsRef, (snap) => {
                const list = document.getElementById('item-list');
                list.innerHTML = '';
                snap.forEach(d => {
                    const item = d.data();
                    if (item.status === 'completed') return;
                    const card = document.createElement('div');
                    card.className = "bg-white p-4 rounded-2xl shadow-sm mb-4";
                    card.innerHTML = `
                        <div class="flex space-x-4">
                            <div class="w-20 h-20 rounded-lg flex-shrink-0 flex items-center justify-center text-white font-bold" style="background-color: ${stringToColor(item.title)}">
                                ${item.title[0]}
                            </div>
                            <div class="flex-grow">
                                <h3 class="font-bold text-lg">${item.title}</h3>
                                <p class="text-xs text-gray-400 mb-1">出品者: ${item.ownerId.substring(0,8)}...</p>
                                <p class="text-sm text-gray-600 line-clamp-2">${item.description}</p>
                            </div>
                        </div>
                        <div class="mt-4 flex justify-between items-center">
                            <span class="text-xs bg-orange-100 text-orange-600 px-2 py-1 rounded">${item.category}</span>
                            ${item.ownerId === userId ? 
                                `<button onclick="deleteItem('${d.id}')" class="text-red-400 text-sm">削除</button>` :
                                `<button onclick="startTrade('${d.id}','${item.title}','${item.ownerId}')" class="bg-orange-500 text-white px-6 py-2 rounded-full font-bold shadow-lg">交換を申し込む</button>`
                            }
                        </div>
                    `;
                    list.appendChild(card);
                });
            });
        }

        window.addItem = async function() {
            const title = document.getElementById('add-title').value;
            const desc = document.getElementById('add-desc').value;
            const cat = document.getElementById('add-cat').value;
            if(!title || !desc) return alert("入力してください");
            await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'items'), {
                title, description: desc, category: cat, ownerId: userId, status: 'active', createdAt: new Date()
            });
            document.getElementById('add-title').value = '';
            document.getElementById('add-desc').value = '';
            alert("出品が完了しました！");
        };

        // --- Negotiation & Chat ---
        window.startTrade = async function(itemId, title, sellerId) {
            const negsRef = collection(db, 'artifacts', appId, 'public', 'data', 'negotiations');
            const newNeg = await addDoc(negsRef, {
                itemId, itemTitle: title, sellerId, buyerId: userId, status: 'active',
                messages: [{ sender: 'system', text: "交渉が開始されました！", type: 'text' }]
            });
            switchView('chat', { id: newNeg.id, itemTitle: title, sellerId });
        };

        function loadNegotiations() {
            const negsRef = collection(db, 'artifacts', appId, 'public', 'data', 'negotiations');
            onSnapshot(negsRef, (snap) => {
                const list = document.getElementById('neg-list');
                list.innerHTML = '';
                snap.forEach(d => {
                    const n = d.data();
                    if(n.sellerId !== userId && n.buyerId !== userId) return;
                    const isSeller = n.sellerId === userId;
                    const item = document.createElement('div');
                    item.className = "bg-white p-4 rounded-xl shadow-sm mb-3 flex justify-between items-center cursor-pointer border-l-4 " + (n.status === 'completed' ? 'border-green-500' : 'border-orange-500');
                    item.onclick = () => switchView('chat', { id: d.id, ...n });
                    item.innerHTML = `
                        <div>
                            <p class="font-bold">${n.itemTitle}</p>
                            <p class="text-xs text-gray-500">${isSeller ? '申し込みを受けました' : 'あなたが申し込み中'}</p>
                        </div>
                        <span class="text-xs ${n.status==='completed'?'text-green-500':'text-orange-500'} font-bold">
                            ${n.status === 'completed' ? '取引完了' : '交渉中'}
                        </span>
                    `;
                    list.appendChild(item);
                });
            });
        }

        let chatUnsub = null;
        function initChat(neg) {
            activeNegId = neg.id;
            document.getElementById('chat-title').textContent = neg.itemTitle;
            const completeBtn = document.getElementById('complete-btn');
            
            // 出品者のみ成約ボタンを表示
            if (neg.sellerId === userId && neg.status !== 'completed') {
                completeBtn.classList.remove('hidden');
            } else {
                completeBtn.classList.add('hidden');
            }

            if (chatUnsub) chatUnsub();
            chatUnsub = onSnapshot(doc(db, 'artifacts', appId, 'public', 'data', 'negotiations', neg.id), (d) => {
                if(d.exists()) renderMessages(d.data().messages);
            });
        }

        function renderMessages(msgs) {
            const container = document.getElementById('chat-box');
            container.innerHTML = '';
            msgs.forEach(m => {
                const isMe = m.sender === userId;
                const row = document.createElement('div');
                row.className = `flex w-full mb-4 ${m.sender === 'system' ? 'justify-center' : (isMe ? 'justify-end' : 'justify-start')}`;
                
                if(m.sender === 'system') {
                    row.innerHTML = `<span class="bg-gray-200 text-gray-500 text-[10px] px-3 py-1 rounded-full">${m.text}</span>`;
                } else if(m.type === 'image') {
                    row.innerHTML = `<div class="max-w-[70%] p-1 rounded-xl bg-white shadow-sm border border-gray-100">
                        <div class="w-32 h-32 bg-gray-200 rounded flex items-center justify-center text-gray-400 text-[10px] text-center p-2">
                            [${m.text}]<br>送信されました
                        </div>
                    </div>`;
                } else {
                    row.innerHTML = `<div class="p-3 max-w-[80%] shadow-sm text-sm ${isMe ? 'chat-bubble-me' : 'chat-bubble-other'}">${m.text}</div>`;
                }
                container.appendChild(row);
            });
            container.scrollTop = container.scrollHeight;
        }

        window.sendMsg = async function(type = 'text', customText = null) {
            const input = document.getElementById('chat-input');
            const text = customText || input.value.trim();
            if(!text || !activeNegId) return;
            const negRef = doc(db, 'artifacts', appId, 'public', 'data', 'negotiations', activeNegId);
            await updateDoc(negRef, {
                messages: arrayUnion({ sender: userId, text, type, time: new Date() })
            });
            if(type === 'text') input.value = '';
        };

        window.completeTrade = async function() {
            if(!confirm("本当に交換を完了（成約）しますか？アイテムはマーケットから消えます。")) return;
            const negRef = doc(db, 'artifacts', appId, 'public', 'data', 'negotiations', activeNegId);
            const negSnap = await getDoc(negRef);
            const negData = negSnap.data();
            
            // 1. 交渉ステータスを完了に
            await updateDoc(negRef, { status: 'completed' });
            // 2. アイテムステータスを完了に
            await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'items', negData.itemId), { status: 'completed' });
            // 3. システムメッセージ
            await sendMsg('system', "取引が成立しました！おめでとうございます！");
            alert("取引完了！");
        };

        // --- Helper ---
        function stringToColor(str) {
            let hash = 0;
            for (let i = 0; i < str.length; i++) hash = str.charCodeAt(i) + ((hash << 5) - hash);
            return `hsl(${hash % 360}, 60%, 50%)`;
        }

        window.onload = init;
    </script>
